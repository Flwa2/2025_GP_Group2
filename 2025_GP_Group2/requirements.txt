# ==============================================
# 🧠 WeCast — Full Setup & Run Instructions
# ==============================================
# 1️⃣ Create and activate the virtual environment
#     python -m venv venv
#     venv\Scripts\activate       (Windows)
#     source venv/bin/activate    (Mac/Linux)
#
# 2️⃣ Install all Python dependencies
#     pip install -r requirements.txt
#
# 3️⃣ Install FFmpeg (Required for audio processing)
# --------------------------------------------------
#   • Windows:
#       Download from: https://www.gyan.dev/ffmpeg/builds/
#       Extract the ZIP
#       Set the full path in .env file:
#
#       FFMPEG_PATH=C:/path/to/ffmpeg/bin/ffmpeg.exe
#       FFPROBE_PATH=C:/path/to/ffmpeg/bin/ffprobe.exe
#
#   • Mac:
#       brew install ffmpeg
#
#   • Linux (Ubuntu):
#       sudo apt install ffmpeg
#
#   ⚠️ Very Important:
#   Your .env file MUST contain the correct paths, otherwise
#   audio generation will fail.
#
# --------------------------------------------------
#
# 4️⃣ Environment Variables Required (create .env file next to app.py)
# --------------------------------------------------
# OPENAI_API_KEY=your_key_here
# ELEVENLABS_API_KEY=your_key_here
# FFMPEG_PATH=C:/full/path/to/ffmpeg.exe
# FFPROBE_PATH=C:/full/path/to/ffprobe.exe
#
# --------------------------------------------------
#
# 5️⃣ Start the Flask backend
#     python app.py
# → Backend runs at: http://127.0.0.1:5000
#
# 6️⃣ Setup React (Frontend)
#     cd static/frontend
#     npm install
#     npm install jspdf jspdf-autotable
#
# 7️⃣ Run the React development server
#     npm run dev
# → Frontend runs at: http://localhost:5173
#
# --------------------------------------------------
#
# 8️⃣ Development notes
#   • Flask backend handles all API routes (/api/*)
#   • React frontend handles all UI
#   • Both must run at the same time
#   • Flask root URL redirects automatically to React
#
# --------------------------------------------------
#
# 9️⃣ Optional: Build React for production
#     npm run build
#
# 1️⃣0️⃣ Stop servers
#     Press CTRL+C in each terminal
#
